CREATE TABLE DL_TEST( T_ID INT PRIMARY KEY,
F_NAME VARCHAR(10) NOT NULL ,
L_NAME VARCHAR(10) NOT NULL,
CONTACT INT NOT NULL,
DOB VARCHAR(10) NOT NULL,
BL_GP VARCHAR(10) NOT NULL,
ADDRESS VARCHAR(20) NOT NULL,
STATUS VARCHAR(10) NOT NULL,
TYPE VARCHAR(10) NOT NULL);


CREATE TABLE DL(DL_NO INT PRIMARY KEY,
F_NAME VARCHAR(10) NOT NULL,
L_NAME VARCHAR(10) NOT NULL,
CONTACT INT NOT NULL,
DOB VARCHAR(10) NOT NULL,
DOI VARCHAR(10) NOT NULL,
DOE VARCHAR(10) NOT NULL,
BL_GP VARCHAR(10) NOT NULL,
ADDRESS VARCHAR(20) NOT NULL,
TYPE VARCHAR(10) NOT NULL);

CREATE TABLE OFFICERS(O_ID INT PRIMARY KEY,
O_NAME VARCHAR(10) NOT NULL,
O_CONTACT INT NOT NULL,
DOJ VARCHAR(10) NOT NULL,
O_ADDRESS VARCHAR(20) NOT NULL,
SECTOR VARCHAR(10) NOT NULL,
DESIGNATION VARCHAR(10) NOT NULL);

CREATE TABLE DEPENDENTS(O_ID INT REFERENCES OFFICERS(O_ID),
D_NAME VARCHAR(10) NOT NULL,
D_CONTACT INT NOT NULL,
D_ADDRESS VARCHAR(20) NOT NULL,
D_BG VARCHAR(10) NOT NULL);

CREATE TABLE VEH_REG(REG_NO INT PRIMARY KEY,
DL_NO INT NOT NULL REFERENCES DL(DL_NO),
V_TYPE VARCHAR(10) NOT NULL,
V_BRAND INT NOT NULL,
DOR VARCHAR(10) NOT NULL);

CREATE TABLE EMMISSION_T(E_ID INT PRIMARY KEY,
LVL VARCHAR(10) NOT NULL,
STATS VARCHAR(10) NOT NULL,
E_DATE VARCHAR(10) NOT NULL);

CREATE TABLE CONDUCTS(O_ID INT NOT NULL REFERENCES OFFICERS(O_ID),
REG_NO INT NOT NULL REFERENCES VEH_REG(REG_NO),
E_ID INT NOT NULL REFERENCES EMMISSION_T(E_ID));
